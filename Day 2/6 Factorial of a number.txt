class Solution {
    static ArrayList<Long> factorialNumbers(long n) {
        // code here
        ArrayList<Long> list = new ArrayList<>();
        
        long ans = 1l;
        for(long i = 1l ; i <= n ; i++){
            ans *= i;
            if(ans<=n) list.add(ans);
            else break;
        }
        return list;
    }
   
}
