class Solution{
    static int searchInSorted(int nums[], int N, int k)
    {
        
        // Your code here
        int left = 0;
        int right = N - 1;
        while(left <= right){
            int mid = ( left + right)/2;
            if(nums[mid] == k){
                return 1;
            }else if (nums[mid] > k) right = mid - 1;
            else left = mid+1;
        }
        return -1;
    }
}
