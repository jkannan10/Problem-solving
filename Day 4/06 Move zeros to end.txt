class Solution {
    public void moveZeroes(int[] nums) {
        int cnt = 0;
        int n = nums.length;
        if(n == 1) return;
        for(int i = 0 ; i < n ; i++){
            if(nums[i] == 0) cnt++;
        }
        if(cnt == 0) return;
        for(int i = 0 , j = 0 ; i < n; i++){
            if(nums[i] != 0 ) nums[j++] = nums[i];
        }
        for(int i = 0 , ind = n - cnt ; i < cnt ; i++){
            nums[ind++] = 0;
        }
    }
}
